import os
import logging
from celery import Celery
from celery.signals import task_prerun, task_postrun, task_failure

logger = logging.getLogger(__name__)

# Initialize Celery
celery_app = Celery(
    "ecommerce_ai_suite",
    broker=os.getenv("CELERY_BROKER_URL", "redis://localhost:6379/0"),
    backend=os.getenv("CELERY_RESULT_BACKEND", "redis://localhost:6379/1"),
    include=["celery_worker.tasks"]
)

# Production-optimized configuration
celery_app.conf.update(
    task_serializer="json",
    accept_content=["json"],
    result_serializer="json",
    timezone="UTC",
    enable_utc=True,
    task_track_started=True,
    task_acks_late=True,              # Acknowledge after task completes
    worker_prefetch_multiplier=1,     # Fair task distribution
    task_default_retry_delay=30,
    task_max_retries=3,
    worker_max_tasks_per_child=500,   # Prevent memory leaks
    result_expires=3600 * 24,         # Keep results for 24h
)

# Logging hooks
@task_prerun.connect
def task_prerun_handler(task_id, task, *args, **kwargs):
    logger.info(f"🚀 Task {task.name} [{task_id}] started")

@task_postrun.connect
def task_postrun_handler(task_id, task, *args, **kwargs):
    logger.info(f"✅ Task {task.name} [{task_id}] completed")

@task_failure.connect
def task_failure_handler(task_id, exception, *args, **kwargs):
    logger.error(f"❌ Task {task_id} failed: {str(exception)}")

from fastapi import FastAPI, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from celery.result import AsyncResult
from celery_worker.celery_app import celery_app
from models import WorkflowRequest, WorkflowType
import logging

logging.basicConfig(level=logging.INFO, format="%(levelname)s | %(message)s")
logger = logging.getLogger(__name__)

app = FastAPI(title="Cross-Border E-Commerce AI Suite", version="2.0.0")
app.add_middleware(
    CORSMiddleware, allow_origins=["*"], allow_credentials=True, allow_methods=["*"], allow_headers=["*"]
)

WORKFLOW_TASK_MAP = {
    WorkflowType.MARKETING: "workflow.marketing",
    WorkflowType.CONTENT: "workflow.content",
    WorkflowType.SUPPORT: "workflow.support",
    WorkflowType.ANALYTICS: "workflow.analytics",
    WorkflowType.SALES_IMPROVEMENT: "workflow.sales_improvement",
    WorkflowType.BIZDEV: "workflow.bizdev",
    WorkflowType.SCHEDULER: "workflow.scheduler",
}

@app.post("/api/v1/workflow")
async def trigger_workflow(req: WorkflowRequest):
    task_name = WORKFLOW_TASK_MAP.get(req.workflow_type)
    if not task_name:
        raise HTTPException(400, detail="Invalid workflow_type")
    
    task = celery_app.send_task(task_name, args=[req.inputs])
    return {"job_id": task.id, "status": "PENDING", "message": "Workflow queued successfully"}

@app.get("/api/v1/workflow/{job_id}")
async def get_workflow_status(job_id: str):
    res = AsyncResult(job_id, app=celery_app)
    
    if res.state == "PENDING":
        return {"job_id": job_id, "status": "PENDING", "result": None}
    elif res.state == "PROGRESS":
        return {"job_id": job_id, "status": "RUNNING", "result": res.info}
    elif res.state == "SUCCESS":
        return {"job_id": job_id, "status": "COMPLETED", "result": res.result}
    elif res.state == "FAILURE":
        return {"job_id": job_id, "status": "FAILED", "error": str(res.result)}
    
    return {"job_id": job_id, "status": res.state, "result": None}

@app.get("/health")
async def health_check():
    return {"status": "healthy", "celery_broker": "redis", "registered_workflows": len(WORKFLOW_TASK_MAP)}
