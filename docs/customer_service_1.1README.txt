@app.post("/api/v1/support/ticket")

    """
    Create support ticket with intelligent triage & automation
    Request:
    {
      "ticket_id": "TKT-XXX",
      "customer_email": "user@example.com",
      "inquiry_text": "My order arrived damaged...",
      "order_id": "ORD-12345",
      "item_sku": "PROD-ABC",
      "order_history": {"lifetime_value": 500, "order_count": 3}
    }
    """
    # Run crew synchronously for immediate response (or use background_tasks for async)
    result = run_customer_support_crew(request)
    
    # If escalation needed, trigger human handoff webhook
    if result.get("escalation_flag"):
        background_tasks.add_task(
            notify_human_agent, 
            ticket_id=result["ticket_id"],
            priority=result["sentiment_analysis"]["urgency_level"],
            notes=result["internal_notes"]
        )
    
    return {
        "status": "processed",
        "response": result["generated_response"],
        "rma_label": result.get("logistics_output", {}).get("prepaid_label_url"),
        "escalated": result["escalation_flag"]
    }
