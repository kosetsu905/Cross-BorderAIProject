# tools/support_tools.py
import os

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class CustomerContextTool(BaseTool):
    name: str = "Cross-Platform Customer & Logistics Context Fetcher"
    description: str = "Fetches real-time order logistics status, shipping tracking, refund eligibility, and dispute history for a specific order across Amazon, Shopify, AliExpress, or TikTok."

    def _run(self, order_id: str, platform: str, customer_email: Optional[str] = None) -> dict:
        """
        Inputs:
        - order_id: The unique order identifier
        - platform: "amazon", "shopify", "aliexpress", "tiktok"
        - customer_email: Optional, to check customer refund history/fraud risk
        """
        platform = platform.lower()
        try:
            if platform == "shopify":
                return self._get_shopify_context(order_id, customer_email)
            elif platform == "amazon":
                return self._get_amazon_context(order_id, customer_email)
            elif platform == "aliexpress":
                return self._get_aliexpress_context(order_id, customer_email)
            elif platform == "tiktok":
                return self._get_tiktok_context(order_id, customer_email)
            else:
                return {"error": "Unsupported platform"}
        except Exception as e:
            logger.error(f"CustomerContextTool failed: {e}")
            return {"error": str(e)}

    def _get_shopify_context(self, order_id: str, email: str) -> dict:
        client = ShopifyClient()
        # 1. Fetch Order Details (Logistics)
        order_data = client._make_request(f"orders/{order_id}.json")
        
        # 2. Fetch Refunds
        refunds_data = client._make_request(f"orders/{order_id}/refunds.json")
        
        logistics = order_data.get("order", {}).get("fulfillment_status", "unfulfilled")
        tracking_url = order_data.get("order", {}).get("tracking_url")
        
        # 3. Calculate Refund Rate (Mock logic: in prod, query /customers/{id}/orders.json)
        refund_rate = "N/A" 
        if refunds_data.get("refunds"):
             refund_rate = f"High Risk: {len(refunds_data['refunds'])} refunds on record"

        return {
            "platform": "Shopify",
            "logistics_status": logistics,
            "tracking_url": tracking_url,
            "refund_status": "Refunded" if refunds_data.get("refunds") else "No Refunds",
            "dispute_status": "None", # Shopify disputes usually via Payment Gateway (Stripe/PayPal)
            "customer_risk": refund_rate,
            "raw_data": order_data
        }

    def _get_amazon_context(self, order_id: str, email: str) -> dict:
        # Note: Requires AmazonSPClient implementation
        # client = AmazonSPClient()
        # order_data = client.get_order(order_id)
        
        # Placeholder for Amazon SP-API response structure
        return {

            "tracking_number": "1Z999...",

            "dispute_status": "Open", # e.g., from DisputeStatus
            "a_to_z_claim": "False",
            "note": "Check 'ShippingDetails' for carrier and 'RefundStatus' for financial state."
        }

    def _get_aliexpress_context(self, order_id: str, email: str) -> dict:
        client = AliExpressClient()
        
        # 1. Fetch Logistics (Using Order Query)
        # Endpoint: aliexpress.solution.trade.orderlist.query (filtered by order_id)
        # Note: AliExpress API often requires 'query_order_detail' logic
        
        # 2. Fetch Disputes
        # Endpoint: aliexpress.solution.trade.dispute.query
        # params: {"order_id": order_id}
        
        return {
            "platform": "AliExpress",
            "logistics_status": "WAIT_BUYER_CONFIRM_GOODS",
            "logistics_trace": "Arrived at local sorting center",
            "dispute_status": "Closed", # or "Open", "WaitForSellerResponse"
            "refund_status": "RefundSuccessful",
            "note": "AliExpress disputes often require 'WaitForSellerResponse' action within 72 hours."
        }

    def _get_tiktok_context(self, order_id: str, email: str) -> dict:
        client = TikTokShopClient()
        
        # 1. Order Search (Logistics)
        # Endpoint: order/20230904/orders/search
        payload = {"ids": [order_id]}
        order_data = client._make_request("POST", "/order/20230904/orders/search", body=payload)
        
        # 2. Returns/Refunds Search
        # Endpoint: return/20230904/returns/search
        payload_return = {"order_id": order_id}
        return_data = client._make_request("POST", "/return/20230904/returns/search", body=payload_return)

        order_info = order_data.get("data", {}).get("orders", [{}])[0]
        return_info = return_data.get("data", {}).get("returns", [])

        return {
            "platform": "TikTok Shop",
            "logistics_status": order_info.get("logistics_status"), # e.g., "IN_TRANSIT", "DELIVERED"
            "tracking_no": order_info.get("tracking_no"),
            "refund_status": "No Returns" if not return_info else "Return Requested",
            "dispute_status": "N/A", # TikTok handles refunds via Returns mostly
            "return_reason": return_info[0].get("user_reason") if return_info else None
        }

    1. logistics_status (Current shipping stage)
    2. tracking_url (Link for customer to track)
    3. refund_status/dispute_status (Financial state of the order)

    Pass this data to the resolution phase.
  expected_output: "A structured context dictionary containing logistics, refund, and dispute data for the specific order."

inquiry_resolution:
  description: >
    Based on the retrieved context: "{context_data}", resolve the inquiry: "{inquiry}".
    
    RULES:
    - If logistics_status is "IN_TRANSIT", provide tracking_url and estimated delivery.
    - If dispute_status is "OPEN", explain the current dispute stage and required actions (e.g., "Seller must respond within 72h").
    - If refund_status is "Refunded", provide the refund ID and timeline for funds to appear.
    - If customer has high refund_risk (from context), apply stricter policy checks per PolicyCheckerTool.
    
    Ensure the tone is empathetic but firm on policy.
  expected_output: "A detailed resolution response addressing the logistics or refund issue, citing real-time data from the context."

4. Update crews/support_crew.py
Wire the new tools and tasks into the crew.

# crews/support_crew.py
import os
import yaml
from pydantic import BaseModel, Field
from typing import List
from crewai import Agent, Task, Crew
from crewai_tools import SerperDevTool

# Load configs
with open("config/support_agents.yaml", "r") as f:
    support_agents_config = yaml.safe_load(f)
with open("config/support_tasks.yaml", "r") as f:
    support_tasks_config = yaml.safe_load(f)

# Import New Tools
from tools.support_tools import CustomerContextTool, PolicyCheckerTool

class SupportTicketOutput(BaseModel):
    ticket_summary: str = Field(..., description="Brief summary of the customer inquiry")
    context_retrieved: str = Field(..., description="Summary of logistics/refund data fetched from APIs")
    drafted_response: str = Field(..., description="Final response to the customer")
    internal_notes: str = Field(..., description="Notes for the support team (e.g., 'High refund risk', 'Dispute open')")

def run_support_crew(inputs: dict) -> dict:
    os.environ.setdefault("OPENAI_MODEL_NAME", "gpt-4o-mini")
    
    # Tools
    serper_tool = SerperDevTool()
    context_tool = CustomerContextTool()
    policy_tool = PolicyCheckerTool()
    
    # Agents
    # Agent 1: Context Retriever (Can be the same agent, but conceptually distinct step)
    support_agent = Agent(
        config=support_agents_config["senior_support_agent"],
        tools=[context_tool, policy_tool, serper_tool]
    )
    
    # Tasks
    # Task 1: Context Retrieval (Hidden task to gather data)
    context_task = Task(
        config=support_tasks_config["context_retrieval"],
        agent=support_agent,
        output_pydantic=None # Just internal context
    )
    
    # Task 2: Resolution (Uses context from Task 1)
    resolution_task = Task(
        config=support_tasks_config["inquiry_resolution"],
        agent=support_agent,
        context=[context_task],
        output_pydantic=SupportTicketOutput
    )
    
    # Crew
    support_crew = Crew(
        agents=[support_agent],
        tasks=[context_task, resolution_task],
        verbose=True,
        memory=True
    )
    
    result = support_crew.kickoff(inputs=inputs)
    return result.pydantic.model_dump()
