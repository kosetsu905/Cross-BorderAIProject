4. crews/scheduler_crew.py
import os
import yaml
from pydantic import BaseModel, Field
from typing import List, Dict
from crewai import Agent, Task, Crew
from crewai_tools import SerperDevTool, ScrapeWebsiteTool
from tools.scheduler_tools import TimezoneHolidayTool, ConflictCheckerTool, NotificationRouterTool

# Load configs
with open("config/scheduler_agents.yaml", "r") as f:
    scheduler_agents_config = yaml.safe_load(f)
with open("config/scheduler_tasks.yaml", "r") as f:
    scheduler_tasks_config = yaml.safe_load(f)

# Structured Output (Pydantic v2 compatible)
class ScheduledEvent(BaseModel):
    event_name: str = Field(..., description="Name of the scheduled event/campaign")
    region: str = Field(..., description="Target market/region")
    start_utc: str = Field(..., description="Start time in UTC (ISO 8601)")
    end_utc: str = Field(..., description="End time in UTC (ISO 8601)")
    local_time_note: str = Field(..., description="Optimal local time & timezone context")
    priority: str = Field(..., description="Priority level (High/Medium/Low)")

class ReminderTrigger(BaseModel):
    trigger_offset: str = Field(..., description="Time before event (e.g., T-7d, T-1d)")
    channel: str = Field(..., description="Notification channel (Email, Slack, SMS, Calendar)")
    message_preview: str = Field(..., description="Preview of reminder content")
    escalation_rule: str = Field(..., description="Action if unacknowledged")

class SchedulerOutput(BaseModel):
    optimized_schedule: List[ScheduledEvent] = Field(..., description="Final conflict-free event schedule")
    regional_calendar_notes: str = Field(..., description="Timezone, holiday, and peak season context")
    reminder_sequence: List[ReminderTrigger] = Field(..., description="Multi-channel reminder & escalation plan")
    conflict_resolution_log: List[str] = Field(..., description="Resolved conflicts & adjustment rationale")
    export_instructions: str = Field(..., description="Steps to export to Google Calendar/Outlook/ICS")

def run_scheduler_crew(inputs: dict) -> dict:
    """Callable wrapper for FastAPI orchestration."""
    os.environ.setdefault("OPENAI_MODEL_NAME", "gpt-4o-mini")
    
    # Tools
    tz_holiday_tool = TimezoneHolidayTool()
    conflict_tool = ConflictCheckerTool()
    notif_tool = NotificationRouterTool()
    
    # Agents
    calendar_mgr = Agent(config=scheduler_agents_config["global_calendar_manager"], tools=[tz_holiday_tool])
    campaign_planner = Agent(config=scheduler_agents_config["campaign_alignment_planner"], tools=[SerperDevTool()])
    conflict_resolver = Agent(config=scheduler_agents_config["conflict_resolution_optimizer"], tools=[conflict_tool])
    notif_coord = Agent(config=scheduler_agents_config["notification_coordinator"], tools=[notif_tool])
    
    # Tasks & Context Wiring
    t1 = Task(config=scheduler_tasks_config["timezone_holiday_mapping"], agent=calendar_mgr)
    t2 = Task(config=scheduler_tasks_config["campaign_event_alignment"], agent=campaign_planner, context=[t1])
    t3 = Task(config=scheduler_tasks_config["conflict_resolution_optimization"], agent=conflict_resolver, context=[t2])
    t4 = Task(
        config=scheduler_tasks_config["notification_reminder_setup"],
        agent=notif_coord,
        context=[t3],
        output_pydantic=SchedulerOutput
    )
    
    # Crew Assembly
    scheduler_crew = Crew(
        agents=[calendar_mgr, campaign_planner, conflict_resolver, notif_coord],
        tasks=[t1, t2, t3, t4],
        verbose=False,
        memory=True
    )
    
    # Execution & Structured Return
    result = scheduler_crew.kickoff(inputs=inputs)
    return result.pydantic.model_dump() if hasattr(result.pydantic, "model_dump") else result.pydantic.dict()
