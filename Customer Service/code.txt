 3. crews/support_crew.py

import os
import yaml
from pydantic import BaseModel, Field
from typing import List
from crewai import Agent, Task, Crew
from crewai_tools import SerperDevTool, ScrapeWebsiteTool, WebsiteSearchTool

# Load configs
with open("config/support_agents.yaml", "r") as f:
    support_agents_config = yaml.safe_load(f)
with open("config/support_tasks.yaml", "r") as f:
    support_tasks_config = yaml.safe_load(f)

# Structured Output (Pydantic v2 compatible)
class SupportTicketOutput(BaseModel):
    ticket_summary: str = Field(..., description="Brief summary of the customer inquiry")
    resolution_steps: List[str] = Field(..., description="Step-by-step resolution or troubleshooting guide")
    drafted_response: str = Field(..., description="Final, polished response ready to send to the customer")
    qa_notes: str = Field(..., description="Quality assurance feedback and compliance checks")
    recommended_follow_up: str = Field(..., description="Suggested next steps or follow-up timing")

def run_support_crew(inputs: dict) -> dict:
    """Callable wrapper for FastAPI orchestration."""
    os.environ.setdefault("OPENAI_MODEL_NAME", "gpt-4o-mini")
    
    # Tools
    serper_tool = SerperDevTool()
    scrape_tool = ScrapeWebsiteTool()
    web_search_tool = WebsiteSearchTool()
    
    # Agents
    support_agent = Agent(
        config=support_agents_config["senior_support_agent"],
        tools=[serper_tool, scrape_tool, web_search_tool],
        allow_delegation=True
    )
    qa_agent = Agent(
        config=support_agents_config["support_qa_specialist"],
        allow_delegation=False
    )
    
    # Tasks & Context Wiring
    resolution_task = Task(
        config=support_tasks_config["inquiry_resolution"],
        agent=support_agent
    )
    qa_task = Task(
        config=support_tasks_config["quality_assurance_review"],
        agent=qa_agent,
        context=[resolution_task],
        output_pydantic=SupportTicketOutput
    )
    
    # Crew Assembly
    support_crew = Crew(
        agents=[support_agent, qa_agent],
        tasks=[resolution_task, qa_task],
        verbose=False,
        memory=True
    )
    
    # Execution & Structured Return
    result = support_crew.kickoff(inputs=inputs)
    return result.pydantic.model_dump() if hasattr(result.pydantic, "model_dump") else result.pydantic.dict()
